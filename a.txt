let c = import("std/class.tl");

let Vehicle = c.create_class({
  init: (pos) => {
    self.pos = pos;
  },
  drive: (dist) => {
    self.pos += dist;
    return self;
  },
});

let Car = c.extend_class(Vehicle, {
  init: (pos, speed) => {
    super.init(pos);
    self.speed = speed;
  },
  drive: (dist) => {
    super.drive(dist * self.speed);
    return self;
  },
});

func main() {
  print(Car.new(0, 2));
  print(Vehicle.new(0));
}
